int write (
  1. List<int> list
)

Writes a list of bytes to the serial port. Returns the number of bytes written on success,

Implementation

int write(List<int> list) {
  _checkStatus();
  if (list.isEmpty) {
    return 0;
  }
  var ptr = allocate<Int8>(count: list.length);
  try {
    var index = 0;
    for (var i in list) {
      ptr.elementAt(index++).value = i;
    }
    var result = _nativeWrite(_serialHandle, ptr, list.length);
    return _checkError(result);
  } finally {
    free(ptr);
  }
}
